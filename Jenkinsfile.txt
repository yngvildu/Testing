pipeline {
    agent any

    stages {
        stage('Setup Environment') {
            steps {
                bat '''
                python -m venv venv
                venv\\Scripts\\pip install -r requirements.txt
                '''
            }
        }

        stage('Run Tests') {
            steps {
                bat "venv\\Scripts\\pytest \"Test Automation/Amazon\" --junitxml=results.xml"
            }
        }
    }

    post {
        always {
            junit allowEmptyResults: true, testResults: 'results.xml'
        }
    }
}
